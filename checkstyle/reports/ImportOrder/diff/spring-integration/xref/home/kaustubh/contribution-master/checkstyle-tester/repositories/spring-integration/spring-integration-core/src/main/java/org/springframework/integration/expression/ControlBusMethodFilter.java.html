17  package org.springframework.integration.expression;
18  
19  import java.lang.reflect.Method;
20  import java.util.ArrayList;
21  import java.util.List;
22  
23  import org.springframework.context.Lifecycle;
24  import org.springframework.core.annotation.AnnotationFilter;
25  import org.springframework.core.annotation.MergedAnnotations;
26  import org.springframework.core.annotation.RepeatableContainers;
27  import org.springframework.expression.MethodFilter;
28  import org.springframework.integration.core.Pausable;
29  import org.springframework.jmx.export.annotation.ManagedAttribute;
30  import org.springframework.jmx.export.annotation.ManagedOperation;
31  import org.springframework.util.CustomizableThreadCreator;
32  import org.springframework.util.ReflectionUtils;
33  
34  /**
35   * SpEL {@link MethodFilter} to restrict method invocations to:
36   * <ul>
37   *     <li> {@link Pausable} or {@link Lifecycle} components
38   *     <li> {@code get}, {@code set} and {@code shutdown} methods of {@link CustomizableThreadCreator}
39   *     <li> methods with {@link ManagedAttribute} and {@link ManagedOperation} annotations
40   * </ul>
41   * This class isn't designed for target applications and typically is used from {@code ExpressionControlBusFactoryBean}.
42   *
43   * @author Mark Fisher
44   * @author Artem Bilan
45   * @author Gary Russell
46   *
47   * @since 4.0
48   */
49  public class ControlBusMethodFilter implements MethodFilter {
50  
51  	@Override
52  	public List<Method> filter(List<Method> methods) {
53  		List<Method> supportedMethods = new ArrayList<>();
54  		for (Method method : methods) {
55  			if (accept(method)) {
56  				supportedMethods.add(method);
57  			}
58  		}
59  		return supportedMethods;
60  	}
61  
62  	private boolean accept(Method method) {
63  		Class<?> declaringClass = method.getDeclaringClass();
64  		String methodName = method.getName();
65  		if ((Pausable.class.isAssignableFrom(declaringClass) || Lifecycle.class.isAssignableFrom(declaringClass))
66  				&& ReflectionUtils.findMethod(Pausable.class, methodName, method.getParameterTypes()) != null) {
67  			return true;
68  		}
69  
70  		if (CustomizableThreadCreator.class.isAssignableFrom(declaringClass)
71  				&& (methodName.startsWith("get")
72  				|| methodName.startsWith("set")
73  				|| methodName.startsWith("shutdown"))) {
74  			return true;
75  		}
76  
77  		MergedAnnotations mergedAnnotations =
78  				MergedAnnotations.from(method, MergedAnnotations.SearchStrategy.TYPE_HIERARCHY,
79  						RepeatableContainers.none(), AnnotationFilter.PLAIN);
80  
81  		return mergedAnnotations.get(ManagedAttribute.class).isPresent()
82  				|| mergedAnnotations.get(ManagedOperation.class).isPresent();
83  	}
84  
85  }
