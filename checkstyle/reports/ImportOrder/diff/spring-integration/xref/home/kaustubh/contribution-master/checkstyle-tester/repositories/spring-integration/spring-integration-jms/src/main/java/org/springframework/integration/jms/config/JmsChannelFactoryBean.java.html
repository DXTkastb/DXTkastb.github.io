374 	@Override
375 	protected AbstractJmsChannel createInstance() {
376 		this.initializeJmsTemplate();
377 		if (this.messageDriven) {
378 			this.listenerContainer = createContainer();
379 			SubscribableJmsChannel subscribableJmsChannel =
380 					new SubscribableJmsChannel(this.listenerContainer, this.jmsTemplate);
381 			subscribableJmsChannel.setMaxSubscribers(this.maxSubscribers);
382 			this.channel = subscribableJmsChannel;
383 		}
384 		else {
385 			Assert.isTrue(!Boolean.TRUE.equals(this.pubSubDomain),
386 					"A JMS Topic-backed 'publish-subscribe-channel' must be message-driven.");
387 			PollableJmsChannel pollableJmschannel = new PollableJmsChannel(this.jmsTemplate);
388 			JavaUtils.INSTANCE
389 				.acceptIfNotNull(this.messageSelector, pollableJmschannel::setMessageSelector);
390 			this.channel = pollableJmschannel;
391 		}
392 		if (!CollectionUtils.isEmpty(this.interceptors)) {
393 			this.channel.setInterceptors(this.interceptors);
394 		}
395 		this.channel.setBeanName(this.beanName);
396 		BeanFactory beanFactory = this.getBeanFactory();
397 		JavaUtils.INSTANCE
398 			.acceptIfNotNull(beanFactory, this.channel::setBeanFactory);
399 		this.channel.afterPropertiesSet();
400 		return this.channel;
401 	}
402 
403 	private void initializeJmsTemplate() {
404 		Assert.isTrue(this.destination != null ^ this.destinationName != null,
405 				"Exactly one of destination or destinationName is required.");
406 		JavaUtils.INSTANCE
407 			.acceptIfNotNull(this.destination, this.jmsTemplate::setDefaultDestination)
408 			.acceptIfNotNull(this.destinationName, this.jmsTemplate::setDefaultDestinationName);
409 	}
410 
411 	private AbstractMessageListenerContainer createContainer() {
412 		if (this.containerType == null) {
413 			this.containerType = DefaultMessageListenerContainer.class;
414 		}
415 		AbstractMessageListenerContainer container;
416 		try {
417 			container = this.containerType.newInstance();
418 		}
419 		catch (InstantiationException | IllegalAccessException e) {
420 			throw new IllegalStateException(e);
421 		}
422 		container.setAcceptMessagesWhileStopping(this.acceptMessagesWhileStopping);
423 		container.setAutoStartup(this.autoStartup);
424 		container.setClientId(this.clientId);
425 		container.setConnectionFactory(this.connectionFactory);
426 		JavaUtils.INSTANCE
427 			.acceptIfNotNull(this.destination, container::setDestination)
428 			.acceptIfNotNull(this.destinationName, container::setDestinationName)
429 			.acceptIfNotNull(this.destinationResolver, container::setDestinationResolver);
430 		container.setDurableSubscriptionName(this.durableSubscriptionName);
431 		container.setErrorHandler(this.errorHandler);
432 		container.setExceptionListener(this.exceptionListener);
433 		JavaUtils.INSTANCE
434 			.acceptIfNotNull(this.exposeListenerSession, container::setExposeListenerSession);
435 		container.setMessageSelector(this.messageSelector);
436 		JavaUtils.INSTANCE
437 			.acceptIfNotNull(this.phase, container::setPhase)
438 			.acceptIfNotNull(this.pubSubDomain, container::setPubSubDomain);
439 		container.setSessionAcknowledgeMode(this.sessionAcknowledgeMode);
440 		container.setSessionTransacted(this.sessionTransacted);
441 		container.setSubscriptionDurable(this.subscriptionDurable);
442 		container.setSubscriptionShared(this.subscriptionShared);
443 
444 
445 
446 		if (container instanceof DefaultMessageListenerContainer) {
447 			DefaultMessageListenerContainer dmlc = (DefaultMessageListenerContainer) container;
448 			JavaUtils.INSTANCE
449 				.acceptIfNotNull(this.cacheLevelName, dmlc::setCacheLevelName)
450 				.acceptIfNotNull(this.cacheLevel, dmlc::setCacheLevel)
451 				.acceptIfHasText(this.concurrency, dmlc::setConcurrency)
452 				.acceptIfNotNull(this.concurrentConsumers, dmlc::setConcurrentConsumers)
453 				.acceptIfNotNull(this.maxConcurrentConsumers, dmlc::setMaxConcurrentConsumers)
454 				.acceptIfNotNull(this.idleTaskExecutionLimit, dmlc::setIdleTaskExecutionLimit)
455 				.acceptIfNotNull(this.maxMessagesPerTask, dmlc::setMaxMessagesPerTask);
456 			dmlc.setPubSubNoLocal(this.pubSubNoLocal);
457 			JavaUtils.INSTANCE
458 				.acceptIfNotNull(this.receiveTimeout, dmlc::setReceiveTimeout)
459 				.acceptIfNotNull(this.recoveryInterval, dmlc::setRecoveryInterval);
460 			dmlc.setTaskExecutor(this.taskExecutor);
461 			dmlc.setTransactionManager(this.transactionManager);
462 			JavaUtils.INSTANCE
463 				.acceptIfNotNull(this.transactionName, dmlc::setTransactionName)
464 				.acceptIfNotNull(this.transactionTimeout, dmlc::setTransactionTimeout);
465 		}
466 		else if (container instanceof SimpleMessageListenerContainer) {
467 			SimpleMessageListenerContainer smlc = (SimpleMessageListenerContainer) container;
468 			JavaUtils.INSTANCE
469 				.acceptIfHasText(this.concurrency, smlc::setConcurrency)
470 				.acceptIfNotNull(this.concurrentConsumers, smlc::setConcurrentConsumers);
471 			smlc.setPubSubNoLocal(this.pubSubNoLocal);
472 			smlc.setTaskExecutor(this.taskExecutor);
473 		}
474 		return container;
475 	}

477 	/*
478 	 * SmartLifecycle implementation (delegates to the created channel if message-driven)
479 	 */

481 	@Override
482 	public boolean isAutoStartup() {
483 		return this.channel instanceof SubscribableJmsChannel && ((SubscribableJmsChannel) this.channel).isAutoStartup();
484 	}
485 
486 	@Override
487 	public int getPhase() {
488 		return (this.channel instanceof SubscribableJmsChannel) ?
489 				((SubscribableJmsChannel) this.channel).getPhase() : 0;
490 	}
491 
492 	@Override
493 	public boolean isRunning() {
494 		return this.channel instanceof SubscribableJmsChannel && ((SubscribableJmsChannel) this.channel).isRunning();
495 	}
496 
497 	@Override
498 	public void start() {
499 		if (this.channel instanceof SubscribableJmsChannel) {
500 			((SubscribableJmsChannel) this.channel).start();
501 		}
502 	}
503 
504 	@Override
505 	public void stop() {
506 		if (this.channel instanceof SubscribableJmsChannel) {
507 			((SubscribableJmsChannel) this.channel).stop();
508 		}
509 	}
510 
511 	@Override
512 	public void stop(Runnable callback) {
513 		if (this.channel instanceof SubscribableJmsChannel) {
514 			((SubscribableJmsChannel) this.channel).stop(callback);
515 		}
516 		else {
517 			callback.run();
518 		}
519 	}
520 
521 	@Override
522 	protected void destroyInstance(AbstractJmsChannel instance) {
523 		this.channel.destroy();
524 	}
