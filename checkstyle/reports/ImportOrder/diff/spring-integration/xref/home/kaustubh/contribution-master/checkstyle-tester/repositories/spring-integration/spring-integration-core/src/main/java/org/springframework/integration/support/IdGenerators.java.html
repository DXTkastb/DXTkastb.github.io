17  package org.springframework.integration.support;
18  
19  import java.util.UUID;
20  import java.util.concurrent.atomic.AtomicLong;
21  
22  import org.springframework.util.IdGenerator;
23  
24  
25  /**
26   * Alternative {@link IdGenerator} implementations.
27   *
28   * @author Andy Wilkinson
29   * @author Gary Russell
30   * @since 4.0
31   *
32   */
33  public class IdGenerators {
34  
35  	/**
36  	 * UUID.randomUUID()
37  	 *
38  	 */
39  	public static class JdkIdGenerator implements IdGenerator {
40  
41  		@Override
42  		public UUID generateId() {
43  			return UUID.randomUUID();
44  		}
45  
46  	}
47  
48  	/**
49  	 * Based on the two {@link AtomicLong}s, for {@code topBits} and {@code bottomBits},
50  	 * respectively.
51  	 * Begins with {0, 1}; incremented on each use.
52  	 * <p>
53  	 * Note: after each {@code 2^63} generations, a duplicate {@link UUID}
54  	 * can be returned in a multi-threaded environment, if a second thread sees the old topBits
55  	 * before it is incremented by the thread detecting the bottomBits roll over.
56  	 * The duplicate would be from around the previous rollover and
57  	 * the chance of such a value still being in the system is exceedingly small.
58  	 * If your system might be impacted by this situation, you should choose another {@link IdGenerator}.
59  	 * Also note that there is no persistence, so this generator starts at {0, 1} each time the system
60  	 * is initialized. Therefore, it is not suitable when persisting messages based on their ID; it should
61  	 * only be used when the absolute best performance is required and messages are not persisted.
62  	 */
63  	public static class SimpleIncrementingIdGenerator implements IdGenerator {
64  
65  		private final AtomicLong topBits = new AtomicLong();
66  
67  		private final AtomicLong bottomBits = new AtomicLong();
68  
69  		@Override
70  		public UUID generateId() {
71  			long lowerBits = this.bottomBits.incrementAndGet();
72  			if (lowerBits == 0) {
73  				return new UUID(this.topBits.incrementAndGet(), lowerBits);
74  			}
75  			else {
76  				return new UUID(this.topBits.get(), lowerBits);
77  			}
78  		}
79  
80  	}
81  
82  }
