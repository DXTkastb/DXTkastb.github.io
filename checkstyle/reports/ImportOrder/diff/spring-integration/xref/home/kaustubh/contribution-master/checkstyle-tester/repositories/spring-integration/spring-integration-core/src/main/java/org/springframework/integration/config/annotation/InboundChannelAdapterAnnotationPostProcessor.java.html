17  package org.springframework.integration.config.annotation;
18  
19  import java.lang.annotation.Annotation;
20  import java.lang.reflect.Method;
21  import java.util.List;
22  import java.util.function.Supplier;
23  
24  import org.springframework.beans.factory.NoSuchBeanDefinitionException;
25  import org.springframework.beans.factory.config.ConfigurableListableBeanFactory;
26  import org.springframework.context.annotation.Bean;
27  import org.springframework.core.annotation.AnnotatedElementUtils;
28  import org.springframework.core.annotation.AnnotationUtils;
29  import org.springframework.integration.annotation.InboundChannelAdapter;
30  import org.springframework.integration.config.IntegrationConfigUtils;
31  import org.springframework.integration.core.MessageSource;
32  import org.springframework.integration.endpoint.MethodInvokingMessageSource;
33  import org.springframework.integration.endpoint.SourcePollingChannelAdapter;
34  import org.springframework.integration.util.ClassUtils;
35  import org.springframework.integration.util.MessagingAnnotationUtils;
36  import org.springframework.messaging.MessageHandler;
37  import org.springframework.util.Assert;
38  
39  /**
40   * Post-processor for Methods annotated with {@link InboundChannelAdapter @InboundChannelAdapter}.
41   *
42   * @author Artem Bilan
43   * @author Gary Russell
44   * @author Oleg Zhurakousky
45   *
46   * @since 4.0
47   */
48  public class InboundChannelAdapterAnnotationPostProcessor extends
49  		AbstractMethodAnnotationPostProcessor<InboundChannelAdapter> {
50  
51  
52  	public InboundChannelAdapterAnnotationPostProcessor(ConfigurableListableBeanFactory beanFactory) {
53  		super(beanFactory);
54  	}
55  
56  	@Override
57  	protected String getInputChannelAttribute() {
58  		return AnnotationUtils.VALUE;
59  	}
60  
61  	@Override
62  	public Object postProcess(Object bean, String beanName, Method method, List<Annotation> annotations) {
63  		String channelName =
64  				MessagingAnnotationUtils.resolveAttribute(annotations, AnnotationUtils.VALUE, String.class);
65  		Assert.hasText(channelName, "The channel ('value' attribute of @InboundChannelAdapter) can't be empty.");
66  
67  		MessageSource<?> messageSource = null;
68  		try {
69  			messageSource = createMessageSource(bean, beanName, method);
70  		}
71  		catch (NoSuchBeanDefinitionException e) {
72  			if (this.logger.isDebugEnabled()) {
73  				this.logger.debug("Skipping endpoint creation; "
74  						+ e.getMessage()
75  						+ "; perhaps due to some '@Conditional' annotation.");
76  			}
77  			return null;
78  		}
79  
80  		SourcePollingChannelAdapter adapter = new SourcePollingChannelAdapter();
81  		adapter.setOutputChannelName(channelName);
82  		adapter.setSource(messageSource);
83  		configurePollingEndpoint(adapter, annotations);
84  
85  		return adapter;
86  	}
87  
88  	private MessageSource<?> createMessageSource(Object beanArg, String beanName, Method methodArg) {
89  		MessageSource<?> messageSource = null;
90  		Object bean = beanArg;
91  		Method method = methodArg;
92  		if (AnnotatedElementUtils.isAnnotated(method, Bean.class.getName())) {
93  			Object target = resolveTargetBeanFromMethodWithBeanAnnotation(method);
94  			Class<?> targetClass = target.getClass();
95  			Assert.isTrue(MessageSource.class.isAssignableFrom(targetClass) ||
96  							Supplier.class.isAssignableFrom(targetClass) ||
97  							ClassUtils.isKotlinFaction0(targetClass),
98  					() -> "The '" + this.annotationType + "' on @Bean method " + "level is allowed only for: " +
99  							MessageSource.class.getName() + " or " + Supplier.class.getName() +
100 							(ClassUtils.KOTLIN_FUNCTION_0_CLASS != null
101 									? " or " + ClassUtils.KOTLIN_FUNCTION_0_CLASS.getName()
102 									: "") + " beans");
103 			if (target instanceof MessageSource<?>) {
104 				messageSource = (MessageSource<?>) target;
105 			}
106 			else if (target instanceof Supplier<?>) {
107 				method = ClassUtils.SUPPLIER_GET_METHOD;
108 				bean = target;
109 			}
110 			else if (ClassUtils.KOTLIN_FUNCTION_0_INVOKE_METHOD != null) {
111 				method = ClassUtils.KOTLIN_FUNCTION_0_INVOKE_METHOD;
112 				bean = target;
113 			}
114 		}
115 		if (messageSource == null) {
116 			MethodInvokingMessageSource methodInvokingMessageSource = new MethodInvokingMessageSource();
117 			methodInvokingMessageSource.setObject(bean);
118 			methodInvokingMessageSource.setMethod(method);
119 			String messageSourceBeanName = generateHandlerBeanName(beanName, method);
120 			this.beanFactory.registerSingleton(messageSourceBeanName, methodInvokingMessageSource);
121 			messageSource = (MessageSource<?>) this.beanFactory
122 					.initializeBean(methodInvokingMessageSource, messageSourceBeanName);
123 			if (this.disposables != null) {
124 				this.disposables.add(methodInvokingMessageSource);
125 			}
126 		}
127 		return messageSource;
128 	}
129 
130 	@Override
131 	protected String generateHandlerBeanName(String originalBeanName, Method method) {
132 		return super.generateHandlerBeanName(originalBeanName, method)
133 				.replaceFirst(IntegrationConfigUtils.HANDLER_ALIAS_SUFFIX + "$", ".source");
134 	}
135 
136 	@Override
137 	protected MessageHandler createHandler(Object bean, Method method, List<Annotation> annotations) {
138 		throw new UnsupportedOperationException();
139 	}

141 }
