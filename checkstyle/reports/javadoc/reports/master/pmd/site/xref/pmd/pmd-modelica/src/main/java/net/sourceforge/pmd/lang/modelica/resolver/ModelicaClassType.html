7   /**
8    * A Modelica type that is defined as a <i>class</i> (i.e., class, package, model, etc.).
9    */
10  public interface ModelicaClassType extends ModelicaType, ModelicaDeclaration, SubcomponentResolver {
11      /**
12       * Returns the class specialization (i.e., package, model, function, etc.)
13       */
14      ModelicaClassSpecialization getSpecialization();
15  
16      /**
17       * Returns whether this class is some kind of connector.
18       */
19      boolean isConnectorLike();
20  
21      /**
22       * Returns whether this class is encapsulated.
23       */
24      boolean isEncapsulated();
25  
26      /**
27       * Returns whether this class is partial.
28       */
29      boolean isPartial();
30  
31      /**
32       * Returns the scope defined by this class itself.
33       */
34      ModelicaClassScope getClassScope();
35  }
