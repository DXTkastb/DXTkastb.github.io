79  /**
80   * A {@link BeanPostProcessor} to parse {@link IntegrationFlow} beans and register their
81   * components as beans in the provided
82   * {@link org.springframework.beans.factory.BeanFactory}, if necessary.
83   *
84   * @author Artem Bilan
85   * @author Gary Russell
86   *
87   * @since 5.0
88   */
89  public class IntegrationFlowBeanPostProcessor
90  		implements BeanPostProcessor, ApplicationContextAware, SmartInitializingSingleton {
91  
92  	private ConfigurableApplicationContext applicationContext;
93  
94  	private StringValueResolver embeddedValueResolver;
95  
96  	private ConfigurableListableBeanFactory beanFactory;
97  
98  	private volatile IntegrationFlowContext flowContext;

100 	@Override
101 	public void setApplicationContext(ApplicationContext applicationContext) throws BeansException {
102 		Assert.isInstanceOf(ConfigurableApplicationContext.class, applicationContext,
103 				"To use Spring Integration Java DSL the 'applicationContext' has to be an instance of " +
104 						"'ConfigurableApplicationContext'. Consider using 'GenericApplicationContext' implementation."
105 		);
106 
107 		this.applicationContext = (ConfigurableApplicationContext) applicationContext;
108 		this.beanFactory = this.applicationContext.getBeanFactory();
109 		this.embeddedValueResolver = new EmbeddedValueResolver(this.beanFactory);
110 	}
111 
112 	@Override
113 	public Object postProcessBeforeInitialization(Object bean, String beanName) throws BeansException {
114 		if (bean instanceof StandardIntegrationFlow) {
115 			return processStandardIntegrationFlow((StandardIntegrationFlow) bean, beanName);
116 		}
117 		else if (bean instanceof IntegrationFlow) {
118 			return processIntegrationFlowImpl((IntegrationFlow) bean, beanName);
119 		}
120 		if (bean instanceof IntegrationComponentSpec) {
121 			processIntegrationComponentSpec(beanName, (IntegrationComponentSpec<?, ?>) bean);
122 		}
123 		return bean;
124 	}
125 
126 	@Override
127 	public void afterSingletonsInstantiated() {
128 		for (String beanName : this.beanFactory.getBeanNamesForType(IntegrationFlow.class)) {
129 			if (this.beanFactory.containsBeanDefinition(beanName)) {
130 				String scope = this.beanFactory.getBeanDefinition(beanName).getScope();
131 				if (StringUtils.hasText(scope) && !BeanDefinition.SCOPE_SINGLETON.equals(scope)) {
132 					throw new BeanCreationNotAllowedException(beanName, "IntegrationFlows can not be scoped beans. " +
133 							"Any dependant beans are registered as singletons, meanwhile IntegrationFlow is just a " +
134 							"logical container for them. \n" +
135 							"Consider using [IntegrationFlowContext] for manual registration of IntegrationFlows.");
136 				}
137 			}
138 		}
139 	}

141 	private Object processStandardIntegrationFlow(StandardIntegrationFlow flow, String flowBeanName) { // NOSONAR complexity
142 		String flowNamePrefix = flowBeanName + ".";
143 		if (this.flowContext == null) {
144 			this.flowContext = this.beanFactory.getBean(IntegrationFlowContext.class);
145 		}
146 		boolean useFlowIdAsPrefix = this.flowContext.isUseIdAsPrefix(flowBeanName);
147 		int subFlowNameIndex = 0;
148 		int channelNameIndex = 0;
149 
150 		Map<Object, String> integrationComponents = flow.getIntegrationComponents();
151 		Map<Object, String> targetIntegrationComponents = new LinkedHashMap<>(integrationComponents.size());
152 
153 		for (Map.Entry<Object, String> entry : integrationComponents.entrySet()) {
154 			Object component = entry.getKey();
155 			if (component instanceof ConsumerEndpointSpec) {
156 				ConsumerEndpointSpec<?, ?> endpointSpec = (ConsumerEndpointSpec<?, ?>) component;
157 				MessageHandler messageHandler = endpointSpec.get().getT2();
158 				ConsumerEndpointFactoryBean endpoint = endpointSpec.get().getT1();
159 				String id = endpointSpec.getId();
160 
161 				if (id == null) {
162 					id = generateBeanName(endpoint, flowNamePrefix, entry.getValue(), useFlowIdAsPrefix);
163 				}
164 				else if (useFlowIdAsPrefix) {
165 					id = flowNamePrefix + id;
166 				}
167 
168 				if (noBeanPresentForComponent(messageHandler, flowBeanName)) {
169 					String handlerBeanName = generateBeanName(messageHandler, flowNamePrefix);
170 
171 					registerComponent(messageHandler, handlerBeanName, flowBeanName);
172 					this.beanFactory.registerAlias(handlerBeanName, id + IntegrationConfigUtils.HANDLER_ALIAS_SUFFIX);
173 				}
174 
175 				registerComponent(endpoint, id, flowBeanName);
176 				targetIntegrationComponents.put(endpoint, id);
177 			}
178 			else {
179 				if (noBeanPresentForComponent(component, flowBeanName)) {
180 					if (component instanceof AbstractMessageChannel || component instanceof NullChannel) {
181 						String channelBeanName = ((NamedComponent) component).getComponentName();
182 						if (channelBeanName == null) {
183 							channelBeanName = entry.getValue();
184 							if (channelBeanName == null) {
185 								channelBeanName = flowNamePrefix + "channel" +
186 										BeanFactoryUtils.GENERATED_BEAN_NAME_SEPARATOR + channelNameIndex++;
187 							}
188 						}
189 						registerComponent(component, channelBeanName, flowBeanName);
190 						targetIntegrationComponents.put(component, channelBeanName);
191 					}
192 					else if (component instanceof MessageChannelReference) {
193 						String channelBeanName = ((MessageChannelReference) component).getName();
194 						if (!this.beanFactory.containsBean(channelBeanName)) {
195 							DirectChannel directChannel = new DirectChannel();
196 							registerComponent(directChannel, channelBeanName, flowBeanName);
197 							targetIntegrationComponents.put(directChannel, channelBeanName);
198 						}
199 					}
200 					else if (component instanceof FixedSubscriberChannel) {
201 						FixedSubscriberChannel fixedSubscriberChannel = (FixedSubscriberChannel) component;
202 						String channelBeanName = fixedSubscriberChannel.getComponentName();
203 						if ("Unnamed fixed subscriber channel".equals(channelBeanName)) {
204 							channelBeanName = flowNamePrefix + "channel" +
205 									BeanFactoryUtils.GENERATED_BEAN_NAME_SEPARATOR + channelNameIndex++;
206 						}
207 						registerComponent(component, channelBeanName, flowBeanName);
208 						targetIntegrationComponents.put(component, channelBeanName);
209 					}
210 					else if (component instanceof SourcePollingChannelAdapterSpec) {
211 						SourcePollingChannelAdapterSpec spec = (SourcePollingChannelAdapterSpec) component;
212 						Map<Object, String> componentsToRegister = spec.getComponentsToRegister();
213 						if (!CollectionUtils.isEmpty(componentsToRegister)) {
214 							componentsToRegister.entrySet()
215 									.stream()
216 									.filter(o -> noBeanPresentForComponent(o.getKey(), flowBeanName))
217 									.forEach(o ->
218 											registerComponent(o.getKey(),
219 													generateBeanName(o.getKey(), flowNamePrefix, o.getValue(),
220 															useFlowIdAsPrefix)));
221 						}
222 						SourcePollingChannelAdapterFactoryBean pollingChannelAdapterFactoryBean = spec.get().getT1();
223 						String id = spec.getId();
224 						if (id == null) {
225 							id = generateBeanName(pollingChannelAdapterFactoryBean, flowNamePrefix, entry.getValue(),
226 									useFlowIdAsPrefix);
227 						}
228 						else if (useFlowIdAsPrefix) {
229 							id = flowNamePrefix + id;
230 						}
231 
232 						registerComponent(pollingChannelAdapterFactoryBean, id, flowBeanName);
233 						targetIntegrationComponents.put(pollingChannelAdapterFactoryBean, id);
234 
235 						MessageSource<?> messageSource = spec.get().getT2();
236 						if (noBeanPresentForComponent(messageSource, flowBeanName)) {
237 							String messageSourceId = id + ".source";
238 							if (messageSource instanceof NamedComponent
239 									&& ((NamedComponent) messageSource).getComponentName() != null) {
240 								messageSourceId = ((NamedComponent) messageSource).getComponentName();
241 							}
242 							registerComponent(messageSource, messageSourceId, flowBeanName);
243 						}
244 					}
245 					else if (component instanceof StandardIntegrationFlow) {
246 						String subFlowBeanName =
247 								entry.getValue() != null
248 										? entry.getValue()
249 										: flowNamePrefix + "subFlow" +
250 												BeanFactoryUtils.GENERATED_BEAN_NAME_SEPARATOR + subFlowNameIndex++;
251 						registerComponent(component, subFlowBeanName, flowBeanName);
252 						targetIntegrationComponents.put(component, subFlowBeanName);
253 					}
254 					else if (component instanceof AnnotationGatewayProxyFactoryBean) {
255 						AnnotationGatewayProxyFactoryBean gateway = (AnnotationGatewayProxyFactoryBean) component;
256 						String gatewayId = entry.getValue();
257 
258 						if (gatewayId == null) {
259 							gatewayId = gateway.getComponentName();
260 						}
261 						if (gatewayId == null) {
262 							gatewayId = flowNamePrefix + "gateway";
263 						}
264 
265 						registerComponent(gateway, gatewayId, flowBeanName,
266 								beanDefinition -> {
267 									((AbstractBeanDefinition) beanDefinition)
268 											.setSource(new DescriptiveResource("" + gateway.getObjectType()));
269 								});
270 
271 						targetIntegrationComponents.put(component, gatewayId);
272 					}
273 					else {
274 						String generatedBeanName =
275 								generateBeanName(component, flowNamePrefix, entry.getValue(), useFlowIdAsPrefix);
276 
277 						registerComponent(component, generatedBeanName, flowBeanName);
278 						targetIntegrationComponents.put(component, generatedBeanName);
279 					}
280 				}
281 				else {
282 					Object componentToUse = entry.getKey();
283 					String beanNameToUse = entry.getValue();
284 					if (StringUtils.hasText(beanNameToUse) &&
285 							ConfigurableBeanFactory.SCOPE_PROTOTYPE.equals(
286 									this.beanFactory.getBeanDefinition(beanNameToUse).getScope())) {
287 						this.beanFactory.initializeBean(componentToUse, beanNameToUse);
288 					}
289 					targetIntegrationComponents.put(component, beanNameToUse);
290 				}
291 			}
292 		}
293 
294 		flow.setIntegrationComponents(targetIntegrationComponents);
295 		return flow;
296 	}

298 	/**
299 	 * Only invoked for {@link IntegrationFlow} instances that are not
300 	 * {@link StandardIntegrationFlow}s; typically lambdas. Creates a new
301 	 * {@link StandardIntegrationFlow} with an input channel named {@code beanName.input}
302 	 * and the flow defined by the flow parameter. If the flow is not an
303 	 * {@link IntegrationFlowAdapter} the original, user-provided {@link IntegrationFlow}
304 	 * is wrapped in a proxy and advised with a {@link IntegrationFlowLifecycleAdvice};
305 	 * see its javadocs for more information.
306 	 */
307 	private Object processIntegrationFlowImpl(IntegrationFlow flow, String beanName) {
308 		IntegrationFlowBuilder flowBuilder = IntegrationFlows.from(beanName + ".input");
309 
310 		flow.configure(flowBuilder);
311 
312 		StandardIntegrationFlow target = flowBuilder.get();
313 		processStandardIntegrationFlow(target, beanName);
314 
315 		if (!(flow instanceof IntegrationFlowAdapter)) {
316 			NameMatchMethodPointcutAdvisor integrationFlowAdvice =
317 					new NameMatchMethodPointcutAdvisor(new IntegrationFlowLifecycleAdvice(target));
318 			integrationFlowAdvice.setMappedNames(
319 					"getInputChannel",
320 					"start",
321 					"stop",
322 					"isRunning",
323 					"isAutoStartup",
324 					"getPhase");
325 
326 			ProxyFactory proxyFactory = new ProxyFactory(flow);
327 			proxyFactory.addAdvisor(integrationFlowAdvice);
328 			if (!(flow instanceof SmartLifecycle)) {
329 				proxyFactory.addInterface(SmartLifecycle.class);
330 			}
331 			return proxyFactory.getProxy(this.beanFactory.getBeanClassLoader());
332 		}
333 		else {
334 			return flow;
335 		}
336 	}

338 	private void processIntegrationComponentSpec(String beanName, IntegrationComponentSpec<?, ?> bean) {
339 		Object target = bean.get();
340 
341 		invokeBeanInitializationHooks(beanName, target);
342 
343 		if (bean instanceof ComponentsRegistration) {
344 			Map<Object, String> componentsToRegister = ((ComponentsRegistration) bean).getComponentsToRegister();
345 			if (!CollectionUtils.isEmpty(componentsToRegister)) {
346 
347 				componentsToRegister.entrySet()
348 						.stream()
349 						.filter(component -> noBeanPresentForComponent(component.getKey(), beanName))
350 						.forEach(component ->
351 								registerComponent(component.getKey(),
352 										generateBeanName(component.getKey(), component.getValue())));
353 
354 			}
355 		}
356 	}

358 	private void invokeBeanInitializationHooks(final String beanName, final Object bean) { // NOSONAR complexity
359 		if (bean instanceof Aware) {
360 			if (bean instanceof BeanNameAware) {
361 				((BeanNameAware) bean).setBeanName(beanName);
362 			}
363 			if (bean instanceof BeanClassLoaderAware && this.beanFactory.getBeanClassLoader() != null) {
364 				((BeanClassLoaderAware) bean).setBeanClassLoader(this.beanFactory.getBeanClassLoader()); // NOSONAR
365 			}
366 			if (bean instanceof BeanFactoryAware) {
367 				((BeanFactoryAware) bean).setBeanFactory(this.beanFactory);
368 			}
369 			if (bean instanceof EnvironmentAware) {
370 				((EnvironmentAware) bean).setEnvironment(this.applicationContext.getEnvironment());
371 			}
372 			if (bean instanceof EmbeddedValueResolverAware) {
373 				((EmbeddedValueResolverAware) bean).setEmbeddedValueResolver(this.embeddedValueResolver);
374 			}
375 			if (bean instanceof ResourceLoaderAware) {
376 				((ResourceLoaderAware) bean).setResourceLoader(this.applicationContext);
377 			}
378 			if (bean instanceof ApplicationEventPublisherAware) {
379 				((ApplicationEventPublisherAware) bean).setApplicationEventPublisher(this.applicationContext);
380 			}
381 			if (bean instanceof MessageSourceAware) {
382 				((MessageSourceAware) bean).setMessageSource(this.applicationContext);
383 			}
384 			if (bean instanceof ApplicationContextAware) {
385 				((ApplicationContextAware) bean).setApplicationContext(this.applicationContext);
386 			}
387 		}
388 	}

390 	@SuppressWarnings("unchecked")
391 	private boolean noBeanPresentForComponent(Object instance, String parentBeanName) {
392 		if (instance instanceof NamedComponent) {
393 			String beanName = ((NamedComponent) instance).getBeanName();
394 			if (beanName != null) {
395 				if (this.beanFactory.containsBean(beanName)) {
396 					BeanDefinition existingBeanDefinition = this.beanFactory.getBeanDefinition(beanName);
397 					if (!ConfigurableBeanFactory.SCOPE_PROTOTYPE.equals(existingBeanDefinition.getScope())
398 							&& !instance.equals(this.beanFactory.getBean(beanName))) {
399 
400 						AbstractBeanDefinition beanDefinition =
401 								BeanDefinitionBuilder.genericBeanDefinition((Class<Object>) instance.getClass(),
402 										() -> instance)
403 										.getBeanDefinition();
404 						beanDefinition.setResourceDescription("the '" + parentBeanName + "' bean definition");
405 						throw new BeanDefinitionOverrideException(beanName, beanDefinition, existingBeanDefinition);
406 					}
407 					else {
408 						return false;
409 					}
410 				}
411 				else {
412 					return true;
413 				}
414 			}
415 		}
416 
417 		return !this.beanFactory.getBeansOfType(instance.getClass(), false, false)
418 				.values()
419 				.contains(instance);
420 	}

422 	private void registerComponent(Object component, String beanName) {
423 		registerComponent(component, beanName, null);
424 	}

426 	@SuppressWarnings("unchecked")
427 	private void registerComponent(Object component, String beanName, String parentName,
428 			BeanDefinitionCustomizer... customizers) {
429 
430 		AbstractBeanDefinition beanDefinition =
431 				BeanDefinitionBuilder.genericBeanDefinition((Class<Object>) component.getClass(), () -> component)
432 				.applyCustomizers(customizers)
433 				.getRawBeanDefinition();
434 
435 		if (parentName != null && this.beanFactory.containsBeanDefinition(parentName)) {
436 			AbstractBeanDefinition parentBeanDefinition =
437 					(AbstractBeanDefinition) this.beanFactory.getBeanDefinition(parentName);
438 			beanDefinition.setResource(parentBeanDefinition.getResource());
439 			Object source = parentBeanDefinition.getSource();
440 			if (source instanceof MethodMetadata) {
441 				source = "bean method " + ((MethodMetadata) source).getMethodName();
442 			}
443 			beanDefinition.setSource(source);
444 			this.beanFactory.registerDependentBean(parentName, beanName);
445 		}
446 
447 		((BeanDefinitionRegistry) this.beanFactory).registerBeanDefinition(beanName, beanDefinition);
448 
449 
450 		this.beanFactory.getBean(beanName);
451 	}

453 	private String generateBeanName(Object instance, String prefix) {
454 		return generateBeanName(instance, prefix, null, false);
455 	}
456 
457 	private String generateBeanName(Object instance, String prefix, String fallbackId, boolean useFlowIdAsPrefix) {
458 		if (instance instanceof NamedComponent && ((NamedComponent) instance).getBeanName() != null) {
459 			String beanName = ((NamedComponent) instance).getBeanName();
460 			return useFlowIdAsPrefix
461 					? prefix + beanName
462 					: beanName;
463 		}
464 		else if (fallbackId != null) {
465 			return useFlowIdAsPrefix
466 					? prefix + fallbackId
467 					: fallbackId;
468 		}
469 
470 		String generatedBeanName = prefix;
471 		if (instance instanceof NamedComponent) {
472 			generatedBeanName += ((NamedComponent) instance).getComponentType();
473 		}
474 		else {
475 			generatedBeanName += instance.getClass().getName();
476 		}
477 
478 		String id = generatedBeanName;
479 		int counter = -1;
480 		while (counter == -1 || this.beanFactory.containsBean(id)) {
481 			counter++;
482 			id = generatedBeanName + BeanFactoryUtils.GENERATED_BEAN_NAME_SEPARATOR + counter;
483 		}
484 		return id;
485 	}
