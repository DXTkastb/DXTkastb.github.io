8   import net.sourceforge.pmd.annotation.InternalApi;
9   import net.sourceforge.pmd.lang.symboltable.Scope;
10  import net.sourceforge.pmd.lang.symboltable.ScopedNode;

13  /**
14   * Root interface for all Nodes of the Java AST.
15   */
16  public interface JavaNode extends ScopedNode {
17  
18      /**
19       * Calls back the visitor's visit method corresponding to the runtime type of this Node.
20       *
21       * @param visitor Visitor to dispatch
22       * @param data    Visit data
23       */
24      Object jjtAccept(JavaParserVisitor visitor, Object data);
25  
26  
27      /**
28       * Dispatches the given visitor to the children of this node. This is the default implementation
29       * of {@link JavaParserVisitor#visit(JavaNode, Object)}, to which all other default
30       * implementations for visit methods delegate. Unless visit methods are overridden without calling
31       * {@code super.visit}, the visitor performs a depth-first tree walk.
32       *
33       * <p>The return value of the visit methods called on children are ignored.
34       *
35       * @param visitor Visitor to dispatch
36       * @param data    Visit data
37       *
38       * @deprecated This method is not useful, the logic for combining
39       *     children values should be present on the visitor, not the node
40       */
41      @Deprecated
42      Object childrenAccept(JavaParserVisitor visitor, Object data);
43  
44  
45      @Override
46      JavaNode getChild(int index);
47  
48  
49      @Override
50      JavaNode getParent();
51  
52  
53      @Override
54      Iterable<? extends JavaNode> children();
55  
56  
57      @InternalApi
58      @Deprecated
59      void setScope(Scope scope);
60  
61  }
