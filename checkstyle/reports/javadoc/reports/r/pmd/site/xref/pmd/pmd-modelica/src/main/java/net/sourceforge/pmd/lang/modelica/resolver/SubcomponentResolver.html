7   /**
8    * This interface represents something that, being looked up by some prefix of composite name,
9    * may resolve further name parts. Lexically, this is represented by a dot-notation.
10   *
11   * Please do not confuse this with {@link ModelicaScope} representing <b>lexical</b> scope,
12   * that resolves names as if they are written in the corresponding part of the source file.
13   */
14  public interface SubcomponentResolver {
15      /**
16       * Resolves `<code>name</code> as if resolving subcomponents through the type of base component
17       *
18       * @param state The resolution state
19       * @param name  The name to resolve
20       * @return The resolved declarations
21       */
22      <T extends ResolvableEntity> ResolutionResult<T> safeResolveComponent(Class<T> clazz, ResolutionState state, CompositeName name);
23  }
