5   package net.sourceforge.pmd.lang.modelica.resolver;
6   
7   import java.util.List;
8   
9   /**
10   * A <b>lexical</b> scope of Modelica code.
11   * That is, a component declaration does not have one, it is its type that does (but these may be resolved to multiple
12   * classes or not resolved at all, these classes generally reside in other files, etc.)
13   *
14   * Please do not confuse this with {@link SubcomponentResolver} that represents "view from the outside" on something
15   * possibly looked up from other file via component reference.
16   */
17  public interface ModelicaScope {
18      /**
19       * Returns the declarations that were lexically declared in this scope.
20       */
21      List<ModelicaDeclaration> getContainedDeclarations();
22  
23      /**
24       * Resolves a name as if it is written inside this lexical scope in a file.
25       */
26      <T extends ResolvableEntity> ResolutionResult<T> safeResolveLexically(Class<T> clazz, ResolutionState state, CompositeName name);
27  
28      /**
29       * Returns the parent (i.e., containing) scope.
30       */
31      ModelicaScope getParent();
32  
33      RootScope getRoot();
34  }
